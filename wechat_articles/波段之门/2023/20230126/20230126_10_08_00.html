2.选股公式

A:=(C/INDEXC)*10000;

M13:=MA(A,13);

M21:=MA(A,21);

M1:=MA(C,34);

M2:=MA(C,89);

XG:M13>M21 AND M21>M1 AND M13/M1>1.15 AND C>MA(C,21) AND C>MA(C,55) AND HHV(H,104)/LLV(L,104)>2;

3.强度线

A:=(C/INDEXC)*10000;

M13:MA(A,13);

M34:MA(A,34),COLORLIBLUE,LINETHICK2;

M55:MA(A,55),COLORRED,LINETHICK2;

M89:MA(A,89),COLORYELLOW,DOTLINE;

4.选股公式

A:AMO>2*MA(AMO,20) AND HHV(H,65)/LLV(L,65)>2 AND C<MA(C,21) AND BARSCOUNT(CLOSE)>400;
