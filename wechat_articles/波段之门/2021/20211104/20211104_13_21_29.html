A:=(C/INDEXC)*10000;


M13:=MA(A,13);

M34:=MA(A,34);

M89:=MA(A,89);

M1:=MA(C,55);

M2:=MA(C,89);

XG:M13>M34 AND M34>M55 AND M55>M89 AND M1>M2 AND C>M2;
